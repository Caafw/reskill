// ── API CALL WRAPPER with proper error surfacing ───────────────────────────────

async function callAPI(body){

  let res;

  try {

    res = await fetch('https://api.anthropic.com/v1/messages', {

      method: 'POST',

      headers: buildHeaders(),

      body: JSON.stringify(body)

    });

  } catch(networkErr) {

    // fetch() itself failed — likely CSP block or no network

    console.error('Network error:', networkErr);

    throw new Error('Network error — check your connection or API key setup');

  }

  // Read response as text first so we can handle non-JSON responses

  let text;

  try { text = await res.text(); } catch(e) { throw new Error('Could not read API response'); }

  // Log full response for debugging

  console.log('API status:', res.status, 'body:', text.substring(0,200));

  let parsed;

  try { parsed = JSON.parse(text); } catch(e) {

    throw new Error('API returned non-JSON (status '+res.status+'): '+text.substring(0,100));

  }

  if(!res.ok){

    const msg = parsed?.error?.message || 'HTTP '+res.status;

    throw new Error(msg);

  }

  return parsed;

}

async function loadLessonContent(mi,li){

  if(FREEMIUM){ showFreemiumGate(); return; }

  const btn=document.getElementById(`btn-${mi}-${li}`);

  const contentEl=document.getElementById(`content-${mi}-${li}`);

  if(contentEl.classList.contains('open'))return;

  btn.textContent='Loading';btn.className='load-btn loading-state';

  const mod=S.courseData.modules[mi];

  const lesson=mod.lessons[li];

  const isInstructor=S.role==='Instructor';

  // System prompt is format-agnostic — output format instruction is in user prompt only

  const isDesigner=S.role==='Designer';

  const sysPrompt=isDesigner

    ?`You are converting internal process documentation into training content for "${S.topic}". Use ONLY information from the provided source document. Do not invent or supplement. Level: ${S.level}.`

    :isInstructor

    ?`Subject expert in "${S.topic}", ${S.level} level. Write real subject knowledge for a trainer. No facilitation meta-advice.`

    :`Subject expert in "${S.topic}", ${S.level} level. Teach directly to the student. Real concepts, real examples. Never describe — just teach.`;

  const srcCtx=isDesigner&&S.goals?` Source: "${S.goals.substring(0,600)}"`:'' ;

  const fmtInstructions={

    prose:isDesigner

      ?`Convert process step "${lesson.name}" into training content.${srcCtx} Write 3-4 paragraphs covering: (1) what staff must do, (2) rules/guardrails that apply, (3) a worked example of this process step, (4) common errors. Use source document only. Output: HTML <p> tags only.`

      :isInstructor

      ?`Trainer guide for "${lesson.name}" in ${S.topic}. 3-4 paragraphs of real subject content + 1 worked example + 2 common misconceptions. Subject expert register. Output: HTML <p> tags only.`

      :`Teach "${lesson.name}" in ${S.topic} to a ${S.level} learner. 3-4 paragraphs of real subject content + 1 worked example. No meta-commentary. Output: HTML <p> tags only.`,

    bullets:isDesigner

      ?`Convert "${lesson.name}" into 7 training bullets.${srcCtx} Each bullet must state a specific rule, step, guardrail, or action from the process. Include 1 worked scenario bullet. JSON:{"points":[""]}`

      :isInstructor

      ?`7 trainer subject-matter bullets for "${lesson.name}": real facts/concepts/rules/techniques, 1 worked example. JSON:{"points":[""]}`

      :`7 bullet points teaching real concepts from "${lesson.name}" — definitions, rules, techniques, 1 worked example. JSON:{"points":[""]}`,

    slides:isDesigner

      ?`Convert "${lesson.name}" into 6 trainer talking points.${srcCtx} Each point must be a specific, actionable piece of process knowledge a trainer explains to staff. No generic statements. JSON:{"points":[""]}`

      :isInstructor

      ?`6 trainer slide talking points for "${lesson.name}": real facts/rules/formulas/techniques per slide. JSON:{"points":[""]}`

      :`6 slide points for "${lesson.name}" — each a substantive, concrete piece of knowledge. JSON:{"points":[""]}`

  }[S.format];

  const prompt=`L:"${lesson.name}" M:"${mod.title}" T:${S.topic} ${S.role} ${S.level}\n`+

    fmtInstructions+' '+(S.format==='prose'?'HTML <p> only.':'JSON:{"points":[""]}');

  try{

    const data=await callAPI({model:'claude-sonnet-4-6',max_tokens:500,system:sysPrompt,messages:[{role:'user',content:prompt}]});

    let raw=data.content.find(b=>b.type==='text')?.text||'';

    raw=raw.replace(/```json|```/gi,'').trim();

    renderLessonContent(contentEl,raw,lesson,isInstructor);

  }catch(e){

    console.error('Lesson API error:',e);

    const errMsg=e.message||'Unknown error';

    const isAuth=errMsg.includes('401')||errMsg.includes('authentication')||errMsg.includes('API key');

    if(isAuth||(!isClaudeAi()&&!getApiKey())){

      contentEl.innerHTML='<div style="font-family:var(--sans);font-size:13px;color:#8b1a1a;padding:10px 0;border-top:1px solid #ede9e0"><strong>API key required.</strong> Enter your Anthropic API key in the bar at the top of the page, then click Retry.</div>';

      document.getElementById('api-key-bar').style.display='flex';

    } else {

      contentEl.innerHTML='<div style="font-family:var(--sans);font-size:13px;color:#7a7a7a;padding:10px 0">Could not load: '+errMsg+'. Please try again.</div>';

    }

    contentEl.classList.add('open');

    btn.textContent='Retry';btn.className='load-btn';

    btn.onclick=()=>{contentEl.innerHTML='';contentEl.classList.remove('open');loadLessonContent(mi,li);};

    return;

  }

  btn.textContent='Loaded';btn.className='load-btn loaded';

  btn.onclick=()=>contentEl.classList.toggle('open');

  contentEl.classList.add('open');

}


function renderLessonContent(el,raw,lesson,isInstructor){

  let html='';

  const isDesignerRender=S.role==='Designer';

  const fallback=isDesignerRender

    ?`<p>This training step covers <strong>${lesson.name}</strong>. It is drawn from your source document and covers the specific process, rules, and guardrails your staff must follow.</p><p>Ensure learners can describe the step accurately in their own words and identify the key guardrails that apply before moving on.</p>`

    :isInstructor

    ?`<p>This lesson addresses ${lesson.name}. As the trainer, establish the core concepts of this topic with clarity, using concrete examples from the subject domain. Ensure learners can articulate the key principle in their own words before proceeding.</p><p>Address the most common misconception in this area directly — stating what learners often believe incorrectly and why the accurate account differs. This prevents confusion at later stages of the course.</p>`

    :`<p>This lesson introduces ${lesson.name}. Study the core concept carefully before attempting the exercises. Each idea introduced here will recur in later modules.</p><p>Ensure you can explain the central principle in your own words — this is the most reliable test of genuine understanding rather than surface familiarity.</p>`;

  if(S.format==='prose'){

    const text=raw||fallback;

    html=`<div class="prose-content">${text.split('\n').filter(Boolean).map(p=>p.startsWith('<')?p:`<p>${p}</p>`).join('')}</div>`;

    if(isDesignerRender)html+=`<div class="trainer-aside"><div class="trainer-aside-label">Source Fidelity Note</div>This content is drawn from your process document. Review before deployment to confirm accuracy against your current procedures.</div>`;

    else if(isInstructor)html+=`<div class="trainer-aside"><div class="trainer-aside-label">Instructor Note</div>The above constitutes the subject matter to deliver. Adapt examples to your learners' professional or academic context.</div>`;

  } else {

    let points=[];

    try{const p=JSON.parse(raw);points=p.points||[];}catch(e){}

    if(!points.length)points=isInstructor

      ?[`Core concept: the principal idea the trainer must convey in ${lesson.name}`,`Theoretical foundation: the underlying principle that makes this concept work`,`First worked example: a concrete case the trainer can walk through with learners`,`Second worked example: a contrasting or extended application of the concept`,`Anticipated misconception: what learners often misunderstand, and the correction`,`Discussion prompt: a question that tests whether learners have genuinely understood`,`Assessment indicator: what a correct, competent answer to this topic looks like`]

      :[`Definition: the precise meaning of the central term or concept in this lesson`,`Core principle: the fundamental rule or relationship you must understand`,`Why it matters: the practical or theoretical significance of this concept`,`Worked example: a step-by-step demonstration of the concept in action`,`Common error: the mistake most learners make — and how to avoid it`,`Application: how this concept connects to real situations in the subject`,`Check your understanding: can you explain this concept without looking at your notes?`];

    if(S.format==='bullets'){

      html=`<div class="bullets-content"><ul>${points.map(p=>`<li>${p}</li>`).join('')}</ul></div>`;

      if(isInstructor)html+=`<div class="trainer-aside"><div class="trainer-aside-label">Instructor Note</div>Each point above represents one teaching beat. Develop each with a concrete example when delivering to learners.</div>`;

    } else {

      html=`<div class="slides-content">${points.map((p,i)=>`<div class="slide-item"><div class="slide-n">${isInstructor?`Talking Point ${i+1}`:`Slide ${i+1}`}</div><div class="slide-text">${p}</div></div>`).join('')}</div>`;

      if(isInstructor)html+=`<div class="trainer-aside"><div class="trainer-aside-label">Instructor Note</div>Expand each talking point with a concrete example or demonstration when presenting to learners.</div>`;

    }

  }

  el.innerHTML=html;

}
